¶ function: unsubscribe

Unsubscribe from an event.

Params
type string The type of event to unsubscribe from
fn function The function to remove
Returns
boolean true if unsubscribe was successful, false otherwise 
  obj.unsubscribe = function (type, fn) {
    if (subscribers[type] === undefined){
      return false;
    }
    var s = subscribers[type];
    for(var i = 0; i < s.length; i++) {
      if(s[i] === fn) {
        delete s[i];
        return true;
      }
    }
    return false;
  };
