¶ User Profile Module

Represents a user of Jargn. Collects one person's posts, and shows information about this person, who is following him/her, and other users he/she is following.

  ¶



Get the user whose profile this is

        		
        
  userlib.getUserByName(req.params.username, function (err, theUser) {
    if (err) {
      res.send('User not found', 404);
    }
    else {
      profileUser = theUser;
      var followers;
      userlib.getFollowers(theUser, function(err, arr){
        if(err){}
        else{
          followers = arr;
        }
      });
      var following;
      userlib.getFollowers(theUser, function(err, arr){
        if(err){}
        else{
          following = arr;
        }
      });
      var posts;
      postlib.getPostsByUser(theUser, function (err, arr) {
        if (!err) {
          posts = arr;
        }
      });
      res.render('user', { title    : 'Jargn',
                 otherUser  : theUser,
                 user   : req.session.user,
                 ouFollowers: followers,
                 ouFollowing: following,
                 ouPosts  : posts,
                 isFollowing: userlib.isFollowing});
    }
  });
};
